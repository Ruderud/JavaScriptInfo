3. switch는 if문을 사용해서 재작성할 수 있다. 또한 조건확인시 내부적으로 === (엄격한)일치연산자를 적용해서 비교를 진행한다
    
let age = prompt('나이를 알려주세요.', 18);

switch (age) {
  case 18:
    alert("Won't work"); // prompt 함수는 항상 문자열을 반환하므로, 이 case문을 절대 실행할 수 없다
    break;

  case "18":
    alert("낭랑 18세이시군요!"); //18이 입력되었다면 여기서 이것을 얼럿하고 종료
    break;

  default:
    alert("어떤 case문에도 해당하지 않습니다.");
}

    또한, case문을 묶을 수도있었음.

    화살표함수는 한줄짜리 간단한 함수를 수행할때 사용하면 좋다.
